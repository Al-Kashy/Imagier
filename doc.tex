\newcommand{\writeBig}[1]{
\Huge\bfseries{#1}
}

\newcommand{\writeBigArb}[1]{
\writeBig{\textarabic{#1}}
}

%Write with shadow
\newcommand{\writeShadow}[4]{
	\node[scale={#1}, text=black] at ({#2}, {#3} ) 		{\writeBig{#4}};
	\node[scale={#1}, text=white] at ({#2}, {#3+0.1} )	{\writeBig{#4}}; 
}

%Write with shadow in arabic
\newcommand{\writeShadowArb}[4]{
	\node[scale={#1}, text=black] at ({#2}, {#3} ) 		{\writeBigArb{#4}};
	\node[scale={#1}, text=white] at ({#2}, {#3+0.1} )	{\writeBigArb{#4}}; 
}

%Generate cover page for the volume
\newcommand{\writeCover}[2]{
\newpage
\begin{tikzpicture}
    \node[anchor=center] (image) at (5,5) 		{\includegraphics[width=8.5cm,height=8.5cm]{#1}};
    	\writeShadow{0.7}{5}{7}{Imagier des trois langues}
    	\writeShadow{0.5}{5}{4.5}{Français - Espagnol - Arabe}
	\writeShadow{0.5}{5}{3}{Volume {#2}}  
	\writeShadow{0.5}{5}{1.5}{Al Kashy}
 \end{tikzpicture}
}

%Generate intro page that explain the tree languague
\newcommand{\writeConcept}[1]{
\newpage
\begin{tikzpicture}
    \node[anchor=center] (image) at (5,5) 	{\includegraphics[width=8.5cm,height=8.5cm]{#1}};
    \writeShadow{1}{5}{8}{Français}
    \writeShadow{1}{5}{6}{Espagñol}
    \writeShadowArb{1}{5}{4}{اَلـعَـرَبِـيَّـةُ}
 \end{tikzpicture}
}

%Display a centered image
\newcommand{\paintpage}[1]{
\newpage
\begin{tikzpicture}[remember picture, overlay]
    \node[anchor=center] at (current page.center) { \includegraphics[width=8.5cm, height=8.5cm]{#1}};
\end{tikzpicture}
}

%For the image the French, Spanish, Arabic word
\newcommand{\writepage}[3]{
\newpage
\vspace*{1.5cm}
\begin{center}
	\writeBig{#1} \\
	\vspace*{1.5cm}
	\writeBig{#2} \\
	\vspace*{1.5cm}
	\writeBigArb{#3} \\
\end{center}
}

%The final page is always the same
\newcommand{\displayEndPage}{
\newpage
\begin{tikzpicture}[remember picture, overlay]
    \node[anchor=center, text width=8cm, align=center, text=black] 
    at ([yshift=3.5cm]current page.center) {\writeBig{Crédit Photo :}  };
    \node[anchor=center, text width=8cm, align=center, text=black] 
    at ([yshift=2.5cm]current page.center) {\writeBig{ Dall-E 2 }};

	\node[anchor=center, scale= 0.8 , align=center, text=black] at ([yshift=-0.0cm]current page.center) 
    {\writeBig{mail.mail@gmail.com}};
\end{tikzpicture}
}
